               #   Use @retry(wait=wait_exponential(multiplier=1, min=2, max=10), stop=stop_after_attempt(3))
               #   Strip HTML from content using BeautifulSoup(content, "lxml").get_text()

               #   On other HTTP errors: let tenacity retry, then log warning and continue
               #   Use time.sleep(0.5) between companies (conservative rate limiting)
