       **DiscoveryResult dataclass:**
       ```python
       from dataclasses import dataclass, field

       @dataclass
       class SourceResult:
           source_id: str
           new_jobs: int = 0
           duplicate_jobs: int = 0
           error: str | None = None

       @dataclass
       class DiscoveryResult:
           sources: list[SourceResult] = field(default_factory=list)
           stale_marked: int = 0
           total_new: int = 0
           total_duplicates: int = 0
       ```

       **build_adapters(config: DiscoveryConfig) -> list[SourceAdapter]:**
       - Instantiate adapters based on non-empty config lists:
         - If config.greenhouse: append GreenhouseAdapter(config.greenhouse)
         - If config.lever: append LeverAdapter(config.lever)
         - Always append HNHiringAdapter(config.hn_months_back) (HN needs no per-company config)
         - If config.wellfound_keywords or config.wellfound_companies: append WellfoundAdapter(config.wellfound_keywords, config.wellfound_companies, config.rate_limit_delay_min, config.rate_limit_delay_max)
       - If `source_filter` is provided (from --source flag), filter to only that adapter by source_id
       - Return list of adapters

       **persist_jobs(session: Session, raw_jobs: list[RawJobDict], source: str) -> tuple[int, int]:**
       - Call get_existing_job_keys(session) ONCE at start (per Pitfall 4 — avoid N+1 queries)
       - For each raw job:
         - normalized = normalize_job(raw, source)
         - is_dup, reason = is_duplicate(normalized.company_slug, normalized.title_normalized, normalized.description_hash, existing_keys)
         - If duplicate: find existing job by company_slug + title_normalized, update its last_seen_at to datetime.utcnow(), increment dedup_count
         - If new: session.add(normalized), session.add(StatusEvent(job_id=normalized.id, status="discovered")), add to existing_keys in-memory, increment new_count
       - session.commit()
       - Return (new_count, dedup_count)

       **update_last_seen(session: Session, company_slug: str, title_normalized: str) -> None:**
       - SELECT NormalizedJob WHERE company_slug AND title_normalized match
       - Update last_seen_at = datetime.utcnow()

       **mark_stale_jobs(session: Session, stale_after_days: int) -> int:**
       - cutoff = datetime.utcnow() - timedelta(days=stale_after_days)
       - SELECT NormalizedJob WHERE last_seen_at < cutoff AND is_stale == False
       - Set is_stale = True for each
       - session.commit()
       - Return count of newly stale jobs

       **Source health tracking** (JSON sidecar file at ~/.config/jobinator/source_health.json per RESEARCH.md Pattern 8):
       ```python
       import json
       from pathlib import Path

       def load_source_health(config_dir: str) -> dict[str, int]:
           path = Path(config_dir) / "source_health.json"
           if path.exists():
               return json.loads(path.read_text())
           return {}

       def save_source_health(config_dir: str, health: dict[str, int]) -> None:
           path = Path(config_dir) / "source_health.json"
           path.parent.mkdir(parents=True, exist_ok=True)
           path.write_text(json.dumps(health, indent=2))

       def fire_health_alerts(health: dict[str, int], console: Console) -> None:
           for source_id, consecutive_zeros in health.items():
               if consecutive_zeros >= 3:
                   console.print(
                       f"[bold yellow]WARNING[/bold yellow] Source '{source_id}' "
                       f"returned 0 results for {consecutive_zeros} consecutive runs. "
                       "Check adapter health or configuration.",
                       style="yellow"
                   )
       ```

       **run_discovery(session: Session, config: DiscoveryConfig, config_dir: str, source_filter: str | None = None) -> DiscoveryResult:**
       - adapters = build_adapters(config) filtered by source_filter if provided
       - health = load_source_health(config_dir)
       - result = DiscoveryResult()
       - For each adapter (sequential, per D-10):
         - try:
           - raw_jobs = adapter.fetch()
           - new_count, dedup_count = persist_jobs(session, raw_jobs, adapter.source_id)
           - result.sources.append(SourceResult(adapter.source_id, new_count, dedup_count))
           - result.total_new += new_count; result.total_duplicates += dedup_count
           - If len(raw_jobs) == 0: health[adapter.source_id] = health.get(adapter.source_id, 0) + 1
           - Else: health[adapter.source_id] = 0
         - except Exception as e:
           - log.warning(f"Source {adapter.source_id} failed: {e}")
           - result.sources.append(SourceResult(adapter.source_id, error=str(e)))
           - (Do NOT increment consecutive_zeros on errors — only on successful 0-result runs)
       - result.stale_marked = mark_stale_jobs(session, config.stale_after_days)
       - save_source_health(config_dir, health)
       - Return result

    1. Update src/jobinator/cli.py to add the `discover` command (per D-07):
       ```python
       from typing import Optional

       @app.command()
       def discover(
           source: Optional[str] = typer.Option(
               None, "--source",
               help="Run a single source adapter (greenhouse, lever, hn_hiring, wellfound)"
           ),
           dry_run: bool = typer.Option(
               False, "--dry-run",
               help="Fetch and normalize but do not persist to database"
           ),
       ) -> None:
           """Discover new jobs from all configured sources."""
           from jobinator.configs.settings import get_discovery_config, get_settings
           from jobinator.db import get_engine, get_session, init_db
           from jobinator.pipelines.discover import fire_health_alerts, load_source_health, run_discovery

           settings = get_settings()
           config = get_discovery_config()

           # Validate --source flag if provided
           valid_sources = {"greenhouse", "lever", "hn_hiring", "wellfound"}
           if source and source not in valid_sources:
               console.print(f"[red]Unknown source '{source}'. Valid: {', '.join(sorted(valid_sources))}[/red]")
               raise typer.Exit(code=1)

           engine = get_engine(settings.database_url)
           init_db(engine)

           with get_session(engine) as session:
               result = run_discovery(session, config, settings.config_dir, source_filter=source)

           # Print Rich summary table
           from rich.table import Table
           table = Table(title="Discovery Results")
           table.add_column("Source", style="cyan")
           table.add_column("New", style="green", justify="right")
           table.add_column("Duplicates", style="yellow", justify="right")
           table.add_column("Status", style="white")

           for src_result in result.sources:
               status = f"[red]ERROR: {src_result.error}[/red]" if src_result.error else "[green]OK[/green]"
               table.add_row(
                   src_result.source_id,
                   str(src_result.new_jobs),
                   str(src_result.duplicate_jobs),
                   status,
               )

           console.print(table)
           console.print(f"\nTotal new: {result.total_new} | Total duplicates: {result.total_duplicates} | Stale marked: {result.stale_marked}")

           # Fire health alerts after summary
           health = load_source_health(settings.config_dir)
           fire_health_alerts(health, console)
       ```
